public class Electrodomestico {
protected  int voltaje;
protected   String ClasificacionEnergetica;
protected   String gama;
protected String marca;
protected String color;
protected String modelo;

   public Electrodomestico(){};
   
   public Electrodomestico(int voltaje,String ClasificacionEnergetica,String gama,String marca,String color,String modelo){
   this.voltaje=220;
   this.ClasificacionEnergetica=ClasificacionEnergetica;
   this.gama=gama;
   this.color=color;
   this.marca=marca;
   this.modelo=modelo;
   }
   //Setter y Getter
   public void setvoltaje(int voltaje){
   this.voltaje=voltaje;
   }
   public int getvoltaje(){
       return this.voltaje;
   }
   public void setClasificacionEnergetica(String ClasificacionEnergetica){
   this.ClasificacionEnergetica=ClasificacionEnergetica;
   }
   public String getClasificacionEnergetica(){
       return this.ClasificacionEnergetica;
   }
   public void setgama(String gama){
   this.gama=gama;
   }
   public String getgama(){
       return this.gama;
   }
   
   public void setmarca(String marca){
   this.marca=marca;
   }
    public String getmarca(){
       return this.marca;
    }
    public void setcolor(String color){
   this.color=color;
   }
    public String getcolor(){
       return this.color;
    }   

    public void setmodelo(String modelo) {
        this.modelo = modelo;
    }

    public String getmodelo() {
        return this.modelo;
    }
    
    
 //Métodos
   public void mostrarDatos(){
       System.out.println("El voltaje es: "+ this.voltaje +"\nLa Clasificación Energetica es: "+ this.ClasificacionEnergetica+ "\nLa Gama es: " +this.gama+"\nLa Marca es: "+this.marca+ "\nEl color es: " +this.color + "\nEl Modelo es: " +this.modelo);
   }
}
